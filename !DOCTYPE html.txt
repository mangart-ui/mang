    <script>
        // ==================== 🎨 본인 일러스트 이미지 주소 설정 ====================
        const images = [
            'https://picsum.photos',
            'https://picsum.photos',
            'https://picsum.photos'
        ];

        const bg1 = document.getElementById('bg1');
        const bg2 = document.getElementById('bg2');
        
        let currentIndex = 0;
        let isBg1Active = true;

        // 초기 이미지 로드
        bg1.style.backgroundImage = `url('${images[0]}')`;
        if(images.length > 1) {
            bg2.style.backgroundImage = `url('${images[0]}')`;
        }

        // 🔄 크로스 디졸브 슬라이드쇼 시스템
        function startSlideshow() {
            if (images.length <= 1) return;

            setInterval(() => {
                currentIndex = (currentIndex + 1) % images.length;

                if (isBg1Active) {
                    bg2.style.backgroundImage = `url('${images[currentIndex]}')`;
                    bg2.classList.add('active');
                    bg1.classList.remove('active');
                } else {
                    bg1.style.backgroundImage = `url('${images[currentIndex]}')`;
                    bg1.classList.add('active');
                    bg2.classList.remove('active');
                }
                
                isBg1Active = !isBg1Active;
            }, 12000); // 12초마다 전환
        }

        // 🖱️ 마우스 패럴랙스 무빙
        function initParallax() {
            let targetX = 0, targetY = 0;
            let currentX = 0, currentY = 0;

            document.addEventListener('mousemove', (e) => {
                targetX = (e.clientX / window.innerWidth - 0.5) * 25;
                targetY = (e.clientY / window.innerHeight - 0.5) * 25;
            });

            function animate() {
                currentX += (targetX - currentX) * 0.05;
                currentY += (targetY - currentY) * 0.05;
                
                bg1.style.transform = `translate(${currentX}px, ${currentY}px)`;
                bg2.style.transform = `translate(${currentX}px, ${currentY}px)`;
                
                requestAnimationFrame(animate);
            }
            animate();
        }

        // 구동
        window.addEventListener('DOMContentLoaded', () => {
            startSlideshow();
            initParallax();
        });
    </script>
